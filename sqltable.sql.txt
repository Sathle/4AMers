CREATE TABLE users(
    user_id number PRIMARY KEY,
    email varchar2(50) NOT NULL,
    username varchar2(30) NOT NULL,
    password varchar2(30) NOT NULL,
    firstname varchar2(30) NOT NULL,
    lastname varchar2(30) NOT NULL,
    date_registered timestamp NOT NULL,
    bio_description varchar2(1000)
);
/
CREATE TABLE pets(
    pet_id number PRIMARY KEY,
    p_name varchar2(30) NOT NULL,
    user_id number NOT NULL,
    breed varchar2(40),
    med_conditions varchar2(500),
    special_needs varchar2(500),
    pet_description varchar2(1000),
    behavior varchar2(500),
    FOREIGN KEY(user_id) REFERENCES users(user_id)
);
/
CREATE TABLE pictures_filePath(
    image_id number PRIMARY KEY,
    user_id number,
    pet_id number,
    image_path varchar2(250) NOT NULL,
    FOREIGN KEY(user_id) REFERENCES users(user_id),
    FOREIGN KEY(pet_id) REFERENCES pets(pet_id)
);

CREATE TABLE inbox(
    message_id number PRIMARY KEY,
    user_id number,
    subject varchar2(250),
    message varchar2(1000),
    date_sent timestamp,
    FOREIGN KEY(user_id) REFERENCES users(user_id)
);

CREATE TABLE service_requests(
    request_id number PRIMARY KEY,
    request_date timestamp,
    pet_id number,
    status varchar2(25),
    sitter_id number,
    poster_id number,
    request_description varchar2(250),
    post_message varchar2(250),
    FOREIGN KEY(sitter_id) REFERENCES users(user_id),
    FOREIGN KEY (poster_id) REFERENCES users(user_id),
    FOREIGN KEY (pet_id) REFERENCES pets(pet_id)
);

CREATE TABLE review(
    review_id number PRIMARY KEY,
    user_id number,
    review_description varchar2(500),
    review_date timestamp,
    review_rating number,
    FOREIGN KEY(user_id) REFERENCES users(user_id)
);

commit;
